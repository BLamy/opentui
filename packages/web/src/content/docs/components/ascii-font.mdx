# ASCIIFont

Display text using ASCII art fonts with multiple font styles available. Great for titles, headers, and decorative text.

## Basic Usage

### Renderable API

```ts example
import { ASCIIFontRenderable, RGBA, createCliRenderer } from "@opentui/core"

const renderer = await createCliRenderer()

const title = new ASCIIFontRenderable(renderer, {
  id: "title",
  text: "OPENTUI",
  font: "tiny",
  color: RGBA.fromInts(255, 255, 255, 255),
})

renderer.root.add(title)
```

### Construct API

```ts example
import { ASCIIFont, createCliRenderer } from "@opentui/core"

const renderer = await createCliRenderer()

renderer.root.add(
  ASCIIFont({
    text: "HELLO",
    font: "block",
    color: "#00FF00",
  }),
)
```

## Available Fonts

OpenTUI includes several ASCII art font styles. This preview renders a short sample of each one so you can compare them side by side:

```ts example
import { ASCIIFont, Box, Text, createCliRenderer, type ASCIIFontName } from "@opentui/core"

const renderer = await createCliRenderer()

function fontSample(label: string, font: ASCIIFontName, text: string, color: string) {
  return Box(
    { flexDirection: "column", gap: 1 },
    Text({ content: label, fg: color }),
    ASCIIFont({
      text,
      font,
      color,
    }),
  )
}

renderer.root.add(
  Box(
    {
      width: "100%",
      height: "100%",
      flexDirection: "row",
      gap: 4,
    },
    Box(
      { flexGrow: 1, flexDirection: "column", gap: 2 },
      fontSample("tiny", "tiny", "A", "#b45309"),
      fontSample("block", "block", "A", "#0f766e"),
      fontSample("shade", "shade", "A", "#7c3aed"),
      fontSample("slick", "slick", "A", "#c2410c"),
    ),
    Box(
      { flexGrow: 1, flexDirection: "column", gap: 2 },
      fontSample("huge", "huge", "A", "#0f766e"),
      fontSample("grid", "grid", "A", "#b45309"),
      fontSample("pallet", "pallet", "A", "#7c3aed"),
    ),
  ),
)
```

## Positioning

Position the ASCII text anywhere on screen:

```ts example
const title = new ASCIIFontRenderable(renderer, {
  id: "title",
  text: "TITLE",
  font: "block",
  color: RGBA.fromHex("#FFFF00"),
  x: 10,
  y: 2,
})
```

## Properties

| Property          | Type                         | Default         | Description                |
| ----------------- | ---------------------------- | --------------- | -------------------------- |
| `text`            | `string`                     | `""`            | Text to display            |
| `font`            | `ASCIIFontName`              | `"tiny"`        | Font style to use          |
| `color`           | `ColorInput \| ColorInput[]` | `"#FFFFFF"`     | Text color(s)              |
| `backgroundColor` | `ColorInput`                 | `"transparent"` | Background color           |
| `selectable`      | `boolean`                    | `true`          | Whether text is selectable |
| `selectionBg`     | `ColorInput`                 | -               | Selection background color |
| `selectionFg`     | `ColorInput`                 | -               | Selection foreground color |
| `x`               | `number`                     | -               | X position offset          |
| `y`               | `number`                     | -               | Y position offset          |

## Example: Welcome Screen

```ts example
import { Box, ASCIIFont, Text, createCliRenderer } from "@opentui/core"

const renderer = await createCliRenderer()

const welcomeScreen = Box(
  {
    width: "100%",
    height: "100%",
    flexDirection: "column",
    alignItems: "center",
    justifyContent: "center",
  },
  ASCIIFont({
    text: "OPENTUI",
    font: "huge",
    color: "#00FFFF",
  }),
  Text({
    content: "Terminal UI Framework",
    fg: "#888888",
  }),
  Text({
    content: "Press any key to continue...",
    fg: "#444444",
  }),
)

renderer.root.add(welcomeScreen)
```

## Dynamic Text

Update the text content dynamically:

```ts example
import { ASCIIFontRenderable, RGBA, createCliRenderer } from "@opentui/core"

const renderer = await createCliRenderer()

const counter = new ASCIIFontRenderable(renderer, {
  id: "counter",
  text: "0",
  font: "block",
  color: RGBA.fromHex("#FF0000"),
})

renderer.root.add(counter)

let count = 0
setInterval(() => {
  count++
  counter.text = count.toString()
}, 1000)
```

## Color Effects

Create gradient-like effects by positioning multiple ASCII fonts:

```ts example
import { Box, ASCIIFont, createCliRenderer } from "@opentui/core"

const renderer = await createCliRenderer()

const gradientTitle = Box(
  {
    width: "100%",
    height: "100%",
  },
  ASCIIFont({
    text: "HELLO",
    font: "block",
    color: "#FF0000",
  }),
  // Overlay with offset for shadow effect
  ASCIIFont({
    text: "HELLO",
    font: "block",
    color: "#880000",
    left: 1,
    top: 1,
  }),
)

renderer.root.add(gradientTitle)
```
